Примеры:

class StringConv {
    String str;

    StringConv(String s) { //конструктор класса
        str = s;
    }

    void stringConv(int index) { //рекурсивный метод
        if(index != str.length()-1) //базовый случай для рекурсии. Пока index не равен последнему элементу массива функция вызывает сама себя
            stringConv(index + 1) //с индексом +1
       
        System.out.print(str.charAt(index));

    }

}

class StringConvTest {
    public static void main(String[] args) {
        StringConv s = new String("Рекурсия");

        s.stringConv(0); //вызвать метод, указав значение начального индекса 0

    }

}
