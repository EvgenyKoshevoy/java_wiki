Вложенные классы

    Делятся на Nested и Inner.

    Nested (Вложенные)

    Должны быть статическими. Определение класса должно быть вложено в другой класс.
    Не дает никакой жесткой связи с классом предком (связь только логическая).

    Пример 1:

    public class Human {
   
        Relations relations;
        String name;
        int age;

        static public enum Relations {    //вложенные enum всегда статические, наличие ключевого слова не обязательно
            DIVORCED, ACTIVE_SEARCH, SINGLE, COMPLEX, IN_LOVE
        }
    }
   
    public class HumanUsage {
        public static void main(String[] args) {
            Human max = new Human.Relations.ACTIVE_SEARCH;
        }
    }

    Пример 2:
   
    public class Human {
        Relations relations;
        String name;
        int age;
       
        static public class Relations {
            private final String name;

            public Relations(final String name) {
                this.name = name;   
            }

            public String getName() {
                return name;
            }
        }
    }

    public class HumanUsage {
        public static void main(String[] args) {
            final Human max = new Human();
            max.relations = new Human.Relations("Single");
        }
    }

    Inner (Внутренние)

    Чтобы создать внутренний класс, необходим instance верхнего класса.

    public class Card {
        private final Srting numer;

        public Card(final String number) {
            this.number = number;
        }

        public String getNumber() {
            return number;
        }
   
        public double withdraw(final double amountToWithDraw) {
            return Account.this.withdraw(amountToWithdraw); //если использовать просто this, то он будет указывать на instance класса Card.
                                                                                                //нам нужен instance класса Account
        }
    }

    Создание класса:
   
    public class Account {
        public static void main(String[] args) {
            final Account myAccount = new Account("123qwer123", "Slava");
            // final Account.Card myCard = new Account.Card("1111 2222 3333 4444"); так выглядело бы создание Nested класса
            final Account.Card myCard = myAccount.new Card("1111 2222 3333 4444");
        }
    }
