SimpleDateFormat

    Является подклассом, производным от класса DateFormat.
    Позволяет определять собственные шаблоны форматирования, чтобы использовать их для отображения сведений о дате и времени.

    Конструктор:
    SimpleDateFormat(String форматируемая_строка)

    Пример:
    SimpleDateFormat sdf = new SimpleDateFormat("dd MMM yyyy hh:mm:ss zzz")

    
public class SimpleDateFormatDemo {
    public static void main(String[] args) {
        Date date = new Date();
        SimpleDateFormat sdf;
        sdf = new SimpleDateFormat("hh:mm:ss");
        System.out.println(sdf.format(date));
        sdf = new SimpleDateFormat("dd MMM yyy hh:mm:ss zzz");
        System.out.println(sdf.format(date));
        sdf = new SimpleDateFormat("E MMM dd yyyy");
        System.out.println(sdf.format(date));
    }
}
    Пример парсинга вводимой даты:
И    И        М
 
public class Solution {

    public static void main(String[] args) throws Exception {
       BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));

       //создать шаблон для вводимой даты
       SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd", Locale.ENGLISH);

       //создать объект date, присвоив ему введенные значения
       Date date = sdf.parse(reader.readLine());

       //создать шаблон для вывода даты
       SimpleDateFormat newDate = new SimpleDateFormat("MMM dd, yyyy", Locale.ENGLISH);

       System.out.println(newDate.format(date).toUpperCase());
       //toUpperCase делает буквы заглавными
    }
}
